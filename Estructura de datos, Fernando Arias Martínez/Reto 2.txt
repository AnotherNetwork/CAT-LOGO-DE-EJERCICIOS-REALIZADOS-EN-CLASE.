#include <stdio.h>

int main() {
    int a = 10;
    int b = 30;
    int resultado;

    int *ptr_a = &a;
    int *ptr_b = &b;

    resultado = *ptr_a + *ptr_b;

    printf("El resultado de la suma es: %d\n", resultado);

    *ptr_a = 30;

    printf("El nuevo valor de a es: %d\n", a);
    printf("El nuevo valor de b es: %d\n", b);

    return 0;
}
